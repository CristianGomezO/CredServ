CREATE TABLE SERVICIOS_DIARIOS (
    ID_SERVICIO INT,
    TURNO VARCHAR2(4),
    CEDULA_CLIENTE VARCHAR2(20), 
    PLACA_AUTOMOTOR VARCHAR2(20), 
    ID_TIPO_VEHICULO NUMBER, 
    ID_TIPO_SERVICIO NUMBER, 
    HORA_ENTRADA VARCHAR2(20), 
    HORA_SALIDA VARCHAR2(20), 
    FECHA_ENTRADA VARCHAR2(20), 
    ID_USUARIO_SERVICIOS INT,
    PRECIO_TOTAL NUMBER,
    TIEMPO_PROMEDIO VARCHAR2(20),
    PORC_DESCUENTO NUMBER,
    TOTAL_DESCUENTO NUMBER,
    PRIMARY KEY (ID_SERVICIO),
	FOREIGN KEY (ID_TIPO_SERVICIO) REFERENCES TIPO_SERVICIO_SERVITECA(ID_TIPO_SERVICIO_SERVITECA),
    FOREIGN KEY (ID_TIPO_VEHICULO) REFERENCES TIPO_VEHICULO_SERVITECA(ID_TIPO_VEHICULO_SERVITECA),
    FOREIGN KEY (ID_USUARIO_SERVICIOS) REFERENCES USUARIO(ID_USUARIO)
    );
    
    
    CREATE SEQUENCE AUTOINCREMENT_SERVICIOSDIARIOS  START WITH 1 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER AUTOINCREMENT_SERVICIOSDIARIOS
BEFORE INSERT ON SERVICIOS_DIARIOS
FOR EACH ROW
BEGIN
:new.ID_SERVICIO := AUTOINCREMENT_SERVICIOSDIARIOS.nextval;
END;
