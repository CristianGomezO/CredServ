CREATE TABLE CLIENTE (	
    ID_CLIENTE INT,
    CEDULA_CONDUCTOR VARCHAR2(20), 
	NOMBRE_CONDUCTOR VARCHAR2(20), 
	APELLIDO_CONDUCTOR VARCHAR2(20),
    PRIMARY KEY (ID_CLIENTE)
   );
   
   CREATE SEQUENCE AUTOINCREMENT_CLIENTE START WITH 1 INCREMENT BY 1;


CREATE OR REPLACE TRIGGER AUTOINCREMENT_CLIENTE
BEFORE INSERT ON CLIENTE
FOR EACH ROW
BEGIN
:new.ID_CLIENTE := AUTOINCREMENT_CLIENTE.nextval;
END;
