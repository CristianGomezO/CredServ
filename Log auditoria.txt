CREATE TABLE LOG_AUDITORIA(
ID_AUDITORIA INT,
ID_USUARIO_LOG INT,
MODULO VARCHAR2(20),
CAMBIO_REALIZADO VARCHAR2(200),
FECHA_CAMBIO VARCHAR2(15),
HORA_CAMBIO VARCHAR2(15),
PRIMARY KEY(ID_AUDITORIA),
FOREIGN KEY (ID_USUARIO_LOG) REFERENCES USUARIO(ID_USUARIO)
);

CREATE SEQUENCE AUTOINCREMENT_LOG  START WITH 1 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER AUTOINCREMENT_LOG
BEFORE INSERT ON LOG_AUDITORIA
FOR EACH ROW
BEGIN
:new.ID_AUDITORIA := AUTOINCREMENT_LOG.nextval;
END;
