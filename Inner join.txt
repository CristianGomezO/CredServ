SELECT TIPO_SERVICIO_SERVITECA.ID_TIPO_SERVICIO_SERVITECA,TIPO_VEHICULO_SERVITECA.ID_TIPO_VEHICULO_SERVITECA ,TIPO_SERVICIO_SERVITECA.TIPO_SERVICIO_SERVITECA, TIPO_VEHICULO_SERVITECA.TIPO_VEHICULO_SERVITECA,TABLA_SERVICIOS_SERVITECA.PRECIO
FROM TABLA_SERVICIOS_SERVITECA INNER JOIN TIPO_SERVICIO_SERVITECA 
ON TABLA_SERVICIOS_SERVITECA.SERVICIOS_ID_TIPO_SERVICIO = TIPO_SERVICIO_SERVITECA.ID_TIPO_SERVICIO_SERVITECA INNER JOIN TIPO_VEHICULO_SERVITECA
ON TIPO_VEHICULO_SERVITECA.ID_TIPO_VEHICULO_SERVITECA = TABLA_SERVICIOS_SERVITECA.SERVICIOS_ID_TIPO_VEHICULO;
